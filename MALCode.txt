; The MAL Code will add number from 1 thourgh 10

       MOVEI 0, R0    ; first summation number (temporary index which will store 0-10) 
       MOVEI 10, R1   ; last summation number, R1 = 10
       MOVEI 0, R2    ; temporary summation (running total) which is initialized as 0
check: BEQ R2, R1, done  ; check if R1 == R2, if R2 == 10, it will branch to Done
sum:   INC R0         ; the first number is increased by 1
       ADD R2, R0, R2 ; R2+R0 = R2 (R2 is a new running total)
       INC R0
BR:    check          ; branch to the address label - check
done:  end            ; halts program
